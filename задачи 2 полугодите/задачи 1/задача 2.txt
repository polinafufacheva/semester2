def is_prime(n):
    """Проверяет, является ли число n простым."""
    if n <= 1:
        return False
    if n == 2:
        return True
    if n % 2 == 0:
        return False
    for i in range(3, int(n**0.5) + 1, 2):
        if n % i == 0:
            return False
    return True

def sum_of_primes_in_interval(a, b):
    """Находит сумму всех простых чисел в интервале от a до b включительно."""
    prime_sum = 0
    for number in range(a, b + 1):
        if is_prime(number):
            prime_sum += number
    return prime_sum

# Запрашиваем у пользователя ввод границ интервала
a = int(input("Введите начало интервала (a): "))
b = int(input("Введите конец интервала (b): "))

# Находим и выводим сумму простых чисел в заданном интервале
result = sum_of_primes_in_interval(a, b)
print(f"Сумма всех простых чисел в интервале от {a} до {b} равна: {result}")
