class Book:
    def __init__(self, title, author):
        self.title = title
        self.author = author

    def get_title(self):
        return self.title

    def get_author(self):
        return self.author

# Пример использования класса Book
book = Book("Война и мир", "Лев Толстой")
print("Название книги:", book.get_title())
print("Автор книги:", book.get_author())
