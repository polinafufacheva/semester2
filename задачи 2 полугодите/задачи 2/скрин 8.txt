import math

class Point:
    def __init__(self, x, y):
        self.x = x
        self.y = y

    def distance_to(self, other_point):
        distance = math.sqrt((self.x - other_point.x)  2 + (self.y - other_point.y)  2)
        return distance

    def move(self, distance, direction):
        if direction == "вверх":
            self.y += distance
        elif direction == "вниз":
            self.y -= distance
        elif direction == "влево":
            self.x -= distance
        elif direction == "вправо":
            self.x += distance
        else:
            print("Неверное направление")

    def is_on_x_axis(self):
        return self.y == 0

    def is_on_y_axis(self):
        return self.x == 0

# Пример использования класса Point
point1 = Point(2, 3)
point2 = Point(5, 7)

distance = point1.distance_to(point2)
print("Расстояние между точками:", distance)

point1.move(3, "вправо")
print("Новые координаты точки 1:", point1.x, point1.y)

print("Точка 1 лежит на оси x?", point1.is_on_x_axis())
print("Точка 1 лежит на оси y?", point1.is_on_y_axis())
