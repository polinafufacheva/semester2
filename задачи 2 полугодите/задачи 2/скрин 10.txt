class Student:
    def __init__(self, name, age):
        self.name = name
        self.age = age
        self.subjects = {}

    def add_subject(self, subject, grade):
        self.subjects[subject] = grade
        print(f"Предмет '{subject}' добавлен для студента {self.name}.")

    def remove_subject(self, subject):
        if subject in self.subjects:
            del self.subjects[subject]
            print(f"Предмет '{subject}' удален для студента {self.name}.")
        else:
            print(f"Предмет '{subject}' не найден у студента {self.name}.")

    def calculate_average_grade(self):
        if not self.subjects:
            return 0
        total_grade = sum(self.subjects.values())
        return total_grade / len(self.subjects)

# Пример использования класса Student
student1 = Student("Иванов", 20)
student1.add_subject("Математика", 4)
student1.add_subject("Физика", 5)
student1.add_subject("Химия", 4)
print("Средний балл студента", student1.name, ":", student1.calculate_average_grade())

student1.remove_subject("Химия")
print("Средний балл студента", student1.name, "после удаления предмета:", student1.calculate_average_grade())
